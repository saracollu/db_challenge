-- 1. Seleziona tutti gli utenti e calcolane l'età (25)
SELECT TIMESTAMPDIFF(YEAR, `birthdate` ,CURDATE()) AS età
FROM `users`;

-- 2. Seleziona tutti i post senza Like (13)
SELECT `posts`.`title`, `posts`.`user_id`, `posts`.`tags`
FROM `posts`
WHERE `posts`.`id` NOT IN (
SELECT `post_id`
FROM `likes` 
);

-- 3. Conta il numero di like per ogni post (165)
SELECT post_id, COUNT(post_id) AS likes
FROM likes
GROUP BY post_id;

-- 4. Ordina gli utenti per il numero di media caricati (25)
SELECT users.username, COUNT(medias.id) AS num_media
FROM users 
LEFT JOIN medias ON users.id = medias.user_id
GROUP BY users.id
ORDER BY num_media DESC;

-- 5. Ordina gli utenti per totale di likes ricevuti nei loro posts (25)
SELECT users.username, COUNT(post_id) AS num_like
FROM users 
LEFT JOIN likes ON users.id = likes.user_id
GROUP BY users.id
ORDER BY num_like DESC;

-- 6. Seleziona tutti i post degli utenti tra i 20 e i 30 anni (49)
SELECT title 
FROM posts
LEFT JOIN users ON posts.user_id = users.id
WHERE TIMESTAMPDIFF(YEAR, users.birthdate, CURDATE()) BETWEEN 20 AND 30;
